{ nr, ... }:
{
  nr.desktop = {
    includes = [
      nr.desktop-niri
      nr.desktop-waybar
      nr.desktop-files
      nr.desktop-terminal
    ];

    provides.apps = {
      homeManager = { pkgs, ... }: {
        home.packages = with pkgs; [ firefox eog evince ];
      };
    };

    provides.xdg = {
      nixos = { pkgs, ... }: {
        xdg.portal.enable = true;
        xdg.portal.extraPortals = [ pkgs.xdg-desktop-portal-gnome ];
      };
    };
  };
}
